def choose_city
  @domain  = params[:domain][:domain]
  @city    = params[:city].to_sym
  @manager = TaskManager.new(@domain,@city)
  logger.info "\n\nChoosing City #{@city} for Domain #{@domain}\n\n"
  logger.info "\n#{@manager.inspect}\n"
end
