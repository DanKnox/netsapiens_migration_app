def initialize(domain,city)
  @tasks  = []
  @tasks << MoveDomainDataTask.new(domain,city)
  @tasks << MoveConferenceDataTask.new(domain,city)
  @tasks << MoveFilesystemDataTask.new(domain,city)
  @tasks << UpdatePhoneConfigTask.new(domain,city)
  @completed_tasks = []

  @tasks.each do |task|

  end
