def execute
  logger "Exporting Sql for #{@domain}\n"
  EXPORT_TABLES.each do |table|
    logger "Sql Output for #{table}\n"
    SSH_CREDENTIALS[PRIMARY_CITY][:model].connection.execute send( table )
  end
end
